17  package com.btmatthews.maven.plugins.emailserver.subethasmtp;
18  
19  import org.subethamail.smtp.MessageHandlerFactory;
20  import org.subethamail.smtp.server.SMTPServer;
21  
22  import com.btmatthews.maven.plugins.emailserver.AbstractMailServer;
23  import com.btmatthews.utils.monitor.Logger;
24  
25  /**
26   * Encapsulates the <a href="http://code.google.com/p/subethasmtp/">SubEtha
27   * SMTP</a> server allowing it to be controlled by a monitor.
28   * 
29   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
30   * @since 1.0.0
31   */
32  public final class SubEthaSMTPMailServer extends AbstractMailServer {
33  
34      /**
35       * Used to control the SubEtha SMTP server.
36       */
37      private SMTPServer server;
38  
39      /**
40       * Start the mail server.
41       * 
42       * @param logger
43       *            Used to log error messages.
44       */
45      public void start(final Logger logger) {
46  	logInfo(logger, "com.btmatthews.maven.plugin.emailserver.subethasmtp.starting");
47  	final MessageHandlerFactory messageHandlerFactory = new SimpleMessageHandlerFactory();
48  	server = new SMTPServer(messageHandlerFactory);
49  	if (isUseSSL()) {
50  	    server.setPort(DEFAULT_SMTPS_PORT + getPortOffset());
51  	    server.setEnableTLS(true);
52  	    server.setRequireTLS(true);
53  	} else {
54  	    server.setPort(DEFAULT_SMTP_PORT + getPortOffset());
55  	}
56  	server.start();
57  	logInfo(logger, "com.btmatthews.maven.plugin.emailserver.subethasmtp.started");
58      }
59  
60      /**
61       * Stop the mail server.
62       * 
63       * @param logger
64       *            Used to log error messages.
65       */
66      public void stop(final Logger logger) {
67  	logInfo(logger, "com.btmatthews.maven.plugin.emailserver.subethasmtp.stopping");
68  	if (server != null) {
69  	    server.stop();
70  	    server = null;
71  	}
72  	logInfo(logger, "com.btmatthews.maven.plugin.emailserver.subethasmtp.stopped");
73      }
74  }
